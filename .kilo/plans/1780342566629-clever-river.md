## FASE 4: Alinear `audio_player.py` con el plan (modelo polling)

### T4.1: Refactorizar AudioPlayer a modelo polling

**Archivo:** `backend/src/services/audio_player.py`

**Cambio fundamental:** Eliminar `_player_loop` thread. Añadir `process_queues(current_gsd)` que el `crewchief_loop` llamará cada tick.

**Nueva interfaz según plan:**
```python
class AudioPlayer:
    def __init__(self, sound_cache, broadcast_callback=None, audio_output=None):
        # Mantener colas, lock, etc.
        self._executor = ThreadPoolExecutor(max_workers=1, thread_name_prefix="audio")
        self._is_message_still_valid = None  # callback

    def set_validator(self, validator: Callable):
        self._is_message_still_valid = validator

    def play_message(self, msg: QueuedMessage):
        """Añade a cola normal."""
        with self._lock:
            prio = msg.priority
            if prio not in self._queue:
                prio = PRIORITY_NORMAL
            self._queue[prio][msg.id] = msg

    def play_message_immediately(self, msg: QueuedMessage):
        """Añade a cola inmediata."""
        with self._lock:
            self._immediate[msg.id] = msg

    def process_queues(self, now=None, current_gsd=None):
        """Llamado por crewchief_loop cada tick.
        Saca UN mensaje, valida contra GSD, reproduce si válido.
        NO bloquea — usa executor para playback.
        """
        now = now or time.time()
        if self._paused:
            if now < self._pause_until:
                return
            self._paused = False

        msg = self._get_next_valid(now, current_gsd)
        if msg:
            self._play_via_executor(msg)

    def _get_next_valid(self, now, gsd):
        """Busca siguiente mensaje válido (inmediata primero, luego normal por prioridad)."""
        with self._lock:
            # Inmediata primero
            for key, msg in list(self._immediate.items()):
                if not msg.is_due(now): continue
                if msg.is_expired(now): self._immediate.pop(key); continue
                if self._is_message_still_valid and gsd:
                    if not self._is_message_still_valid(msg, gsd):
                        self._immediate.pop(key); continue
                self._immediate.pop(key)
                return msg
            # Normal por prioridad descendente
            for prio in sorted(self._queue.keys(), reverse=True):
                q = self._queue[prio]
                for key, msg in list(q.items()):
                    if not msg.is_due(now): continue
                    if msg.is_expired(now): q.pop(key); continue
                    if self._is_message_still_valid and gsd:
                        if not self._is_message_still_valid(msg, gsd):
                            q.pop(key); continue
                    q.pop(key)
                    return msg
        return None

    def _play_via_executor(self, msg):
        """Reproduce en executor thread (no bloquea event loop)."""
        entry = self._cache.get(msg.name)
        if entry is None:
            return
        if self._broadcast:
            self._broadcast(msg)
        self._executor.submit(self._play_wav_blocking, entry.path)

    def _play_wav_blocking(self, path):
        output = self._ensure_audio_output()
        output.play_wav(path, threading.Event())

    def pause_queue(self, seconds: float):
        self._paused = True
        self._pause_until = time.time() + seconds

    def unpause_queue(self):
        self._paused = False
        self._pause_until = 0

    def purge_queues(self):
        """Vacía todas las colas."""
        with self._lock:
            for q in self._queue.values():
                q.clear()
            self._immediate.clear()

    # Eliminar: start(), stop(), _player_loop()
```

**Nota sobre Spotter:** El spotter llama `self.ap.play_spotter_message()` directamente (sin cola). El AudioPlayer debe mantener este método como path separado que reproduce inmediatamente sin pasar por `process_queues()`. Esto es coherente con el plan — el spotter tiene prioridad 20 y bypass a la cola.

**Tests:** Actualizar `test_audio_player.py` — cambiar tests de thread a tests de `process_queues()`.

```python
def get_opponent_key_behind(self, opponents: dict, car_class: str = None) -> Optional[str]:
    """CrewChief: GameStateData.getOpponentKeyBehind()"""
    closest = None
    closest_dist = float('inf')
    pd = self._player_distance  # distance_round_track del jugador
    for key, opp in opponents.items():
        if opp.speed < 0.5 or opp.is_entering_pits:
            continue
        if car_class and opp.vehicle_class != car_class:
            continue
        if opp.distance < pd:
            d = pd - opp.distance
            if d < closest_dist:
                closest, closest_dist = key, d
    # Fallback: lapped traffic ahead
    if not closest:
        for key, opp in opponents.items():
            if opp.speed >= 0.5:
                return key
    return closest
```

```python
    applicable_session_types = [SessionType.PRACTICE, SessionType.QUALIFY, SessionType.RACE]
    applicable_session_phases = [SessionPhase.GREEN, SessionPhase.FULL_COURSE_YELLOW]
```

```python
    _last_fuel: float = 0.0
    _last_lap: int = 0
    _avg_consumption: float = 0.0
    _fuel_ok_after_refuel_announced: bool = False

    def trigger_internal(self, prev, curr):
        # 1. Detectar nueva vuelta → calcular consumo
        # 2. Estimar vueltas restantes con combustible
        # 3. Anunciar combustible bajo (últimas 3 vueltas)
        # 4. Anunciar fin de stint (última vuelta con combustible)

    def _get_avg_consumption(self):

        if len(self._consumption_samples) < 3:
            return 0.0
        recent = self._consumption_samples[-5:]
        return sum(delta for _, delta in recent) / len(recent)

            return 999
        return int(curr.fuel.fuel_left / self._avg_consumption)

    def clear_state(self):
        self._consumption_samples = []
        self._announced_low = False
        self._announced_end_of_stint = False
        self._last_fuel = 0.0
        self._avg_consumption = 0.0
```
